## Task 1: 数据库 Schema

File: backend/src/main/resources/db/migration/V12__publishing_engine.sql

`sql
CREATE TABLE IF NOT EXISTS book_drafts (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    publication_id BIGINT NOT NULL,
    title VARCHAR(255) NOT NULL DEFAULT '未命名谱书',
    subtitle VARCHAR(255) DEFAULT '',
    preface TEXT, epilogue TEXT, style_config JSON,
    status VARCHAR(20) NOT NULL DEFAULT 'draft',
    version INT NOT NULL DEFAULT 1,
    snapshot_revision BIGINT DEFAULT 0,
    created_by BIGINT, updated_by BIGINT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (publication_id) REFERENCES publications(id)
);

CREATE TABLE IF NOT EXISTS book_sheets (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    draft_id BIGINT NOT NULL, sheet_number INT NOT NULL,
    sheet_type VARCHAR(20) NOT NULL DEFAULT 'genealogy',
    layout_data JSON, auto_generated_at DATETIME,
    FOREIGN KEY (draft_id) REFERENCES book_drafts(id) ON DELETE CASCADE,
    UNIQUE KEY uk_draft_sheet (draft_id, sheet_number)
);

CREATE TABLE IF NOT EXISTS book_person_details (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    draft_id BIGINT NOT NULL, person_id VARCHAR(50) NOT NULL,
    biography TEXT, interlinear_notes JSON, seal_style VARCHAR(50),
    user_modified BOOLEAN DEFAULT FALSE, source_fields_hash VARCHAR(32),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (draft_id) REFERENCES book_drafts(id) ON DELETE CASCADE,
    UNIQUE KEY uk_draft_person (draft_id, person_id)
);

CREATE TABLE IF NOT EXISTS sheet_element_overrides (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    sheet_id BIGINT NOT NULL, element_key VARCHAR(64) NOT NULL,
    override_x DECIMAL(10,2), override_y DECIMAL(10,2),
    override_scale DECIMAL(3,2) DEFAULT 1.00,
    override_visible BOOLEAN DEFAULT TRUE, is_manual BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (sheet_id) REFERENCES book_sheets(id) ON DELETE CASCADE,
    UNIQUE KEY uk_sheet_element (sheet_id, element_key)
);

CREATE TABLE IF NOT EXISTS draft_sync_logs (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    draft_id BIGINT NOT NULL, person_id VARCHAR(50) NOT NULL,
    change_type VARCHAR(20) NOT NULL,
    old_values JSON, new_values JSON,
    acknowledged BOOLEAN DEFAULT FALSE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (draft_id) REFERENCES book_drafts(id) ON DELETE CASCADE
);
`
